    ```
    export function formatProfileForDisplay(
      dimension: Dimension,
      profile: ProfileRow<unknown> | null,
      lang: Lang,
    ): string {
      const L = MSG;  // local alias for readability

      // Null or zero-confidence → actionable progress indicator (D-21)
      if (profile === null || profile.confidence === 0) {
        return L.insufficientData[lang](dimension);
      }

      // Build title line
      const confidencePct = Math.round(profile.confidence * 100);
      const title = `${L.sectionTitle[dimension][lang]} (${L.confidence[lang]} ${confidencePct}%)`;
      const lines: string[] = [title, ''];

      // Per-dimension switch-case rendering
      switch (dimension) {
        case 'jurisdictional': {
          const d = profile.data as JurisdictionalProfileData;
          // Render second-person field lines, skipping null/empty per-field
          if (d.physical_location && d.current_country) {
            lines.push(L.fields.jurisdictional[lang].youAreIn(d.physical_location, d.current_country));
          }
          if (d.tax_residency) lines.push(L.fields.jurisdictional[lang].yourTaxResidency(d.tax_residency));
          if (d.residency_status && Array.isArray(d.residency_status) && d.residency_status.length > 0) {
            // Render bullet list per second-person framing
            lines.push(L.fields.jurisdictional[lang].yourResidencyStatuses);
            for (const r of d.residency_status) {
              lines.push(`- ${r.type}: ${r.value}${r.since ? ` (since ${r.since})` : ''}`);
            }
          }
          if (d.next_planned_move && d.planned_move_date) {
            lines.push(L.fields.jurisdictional[lang].yourNextMove(d.next_planned_move.destination ?? String(d.next_planned_move), d.planned_move_date));
          }
          if (d.passport_citizenships && d.passport_citizenships.length > 0) {
            lines.push(L.fields.jurisdictional[lang].yourCitizenships(d.passport_citizenships.join(', ')));
          }
          break;
        }
        case 'capital': { /* parallel structure — FI target + phase + entities */ break; }
        case 'health': { /* per FEATURES.md §2.3 fields */ break; }
        case 'family': { /* per FEATURES.md §2.4 fields */ break; }
      }

      // Staleness note (D-22)
      const STALENESS_MS = 21 * 86_400_000;
      if (Date.now() - profile.lastUpdated.getTime() > STALENESS_MS) {
        const dateStr = profile.lastUpdated.toISOString().slice(0, 10);
        lines.push('', L.staleNote[lang](dateStr));
      }

      return lines.join('\n');
    }
    ```
