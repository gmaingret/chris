**1. Wilson CI (D-15):**
```typescript
export function wilsonCI(hits: number, n: number): { lo: number; hi: number } {
  if (n === 0) return { lo: 0, hi: 0 };
  const z = 1.96;
  const p = hits / n;
  const center = p + (z * z) / (2 * n);
  const margin = z * Math.sqrt((p * (1 - p)) / n + (z * z) / (4 * n * n));
  const denom = 1 + (z * z) / n;
  return {
    lo: Math.max(0, (center - margin) / denom),
    hi: Math.min(1, (center + margin) / denom),
  };
}
```
CRITICAL: Both center AND margin are divided by denom (Pitfall 1 from RESEARCH.md). The correct formula is `lo = (center - margin) / denom`, `hi = (center + margin) / denom`.

**3. Accuracy computation (STAT-03, D-17):**
```typescript
export interface AccuracyResult {
  belowFloor: boolean;
  n: number;
  hits?: number;
  pct?: number;
  ci?: { lo: number; hi: number };
  unverifiable: number;
}

const N_FLOOR = 10;

export function computeAccuracy(rows: StatsRow[]): AccuracyResult {
  // Exclude unverifiable/* and */unknown from denominator (D-04, D-17)
  const scorable = rows.filter(r =>
    r.accuracyClass &&
    !r.accuracyClass.startsWith('unverifiable/') &&
    !r.accuracyClass.endsWith('/unknown')
  );
  const unverifiable = rows.filter(r =>
    r.accuracyClass?.startsWith('unverifiable/') ||
    r.accuracyClass?.endsWith('/unknown')
  ).length;

  const n = scorable.length;
  if (n < N_FLOOR) return { belowFloor: true, n, unverifiable };

  const hits = scorable.filter(r => r.accuracyClass!.startsWith('hit/')).length;
  const pct = Math.round((hits / n) * 100);
  const ci = wilsonCI(hits, n);
  return {
    belowFloor: false, n, hits, pct,
    ci: { lo: Math.round(ci.lo * 100), hi: Math.round(ci.hi * 100) },
    unverifiable,
  };
}
```

```typescript
export interface StatusCounts {
  open: number; due: number; reviewed: number; stale: number;
  openDraft: number; withdrawn: number; abandoned: number;
}

export async function fetchStatusCounts(chatId: bigint): Promise<StatusCounts> {
  const rows = await db
    .select({ status: decisions.status })
    .from(decisions)
    .where(eq(decisions.chatId, chatId));
  const counts: StatusCounts = { open: 0, due: 0, reviewed: 0, stale: 0, openDraft: 0, withdrawn: 0, abandoned: 0 };
  for (const r of rows) {
    if (r.status === 'open-draft') counts.openDraft++;
    else if (r.status in counts) (counts as Record<string, number>)[r.status]++;
  }
  return counts;
}
```
