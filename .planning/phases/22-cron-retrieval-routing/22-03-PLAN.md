<objective>
Extend INTERROGATE mode to detect date references in the user's query and, when the resolved date is older than 7 days, inject the matching episodic summary into the system prompt context block. Date extraction uses a regex/keyword fast-path first and falls back to Haiku only when the heuristic suggests a date reference is probable (keyword presence).

Purpose: RETR-04 is the user-facing payoff for episodic summaries. When Greg asks "what was happening three weeks ago" or "on April 1st", Chris should surface the day's summary as interpretation context — clearly distinguished from verbatim Known Facts (D031). The regex/keyword-first design respects the latency budget (no Haiku call for obvious "last Wednesday" queries) and avoids API cost.

Output: a new `src/chris/modes/date-extraction.ts` helper, a targeted modification to `src/chris/modes/interrogate.ts` that calls the extractor and conditionally injects summary context, and test coverage for both.
</objective>

    ```typescript
    import { anthropic, HAIKU_MODEL } from '../../llm/client.js';
    import { logger } from '../../utils/logger.js';

    /**
     * Keywords that suggest a date reference is probable. When the regex
     * fast-path returns null, the Haiku classifier is only invoked if at least
     * one of these keywords (lowercased) appears in the query. This gates
     * API cost — queries with no date signal whatsoever skip the LLM entirely.
     * Covers EN/FR/RU per D020-D021.
     */
    export const DATE_HEURISTIC_KEYWORDS: readonly string[] = [
      // English
      'ago', 'last', 'yesterday', 'week', 'month', 'year',
      'monday', 'tuesday', 'wednesday', 'thursday', 'friday', 'saturday', 'sunday',
      'january', 'february', 'march', 'april', 'may', 'june',
      'july', 'august', 'september', 'october', 'november', 'december',
      // French
      'hier', 'dernier', 'dernière', 'semaine', 'mois', 'année',
      'il y a',
      'lundi', 'mardi', 'mercredi', 'jeudi', 'vendredi', 'samedi', 'dimanche',
      'janvier', 'février', 'mars', 'avril', 'mai', 'juin',
      'juillet', 'août', 'septembre', 'octobre', 'novembre', 'décembre',
      // Russian
      'назад', 'вчера', 'прошл', 'неделя', 'неделю', 'месяц', 'год',
      'понедельник', 'вторник', 'среда', 'четверг', 'пятница', 'суббота', 'воскресенье',
      'январ', 'феврал', 'март', 'апрел', 'мая', 'июн',
      'июл', 'август', 'сентябр', 'октябр', 'ноябр', 'декабр',
    ] as const;

    /**
     * Attempt to extract an ISO date 'YYYY-MM-DD' directly from the query via
     * regex. Returns a Date at midnight UTC or null.
     */
    function matchIsoDate(text: string): Date | null {
      const m = text.match(/\b(\d{4})-(\d{2})-(\d{2})\b/);
      if (!m) return null;
      const iso = `${m[1]}-${m[2]}-${m[3]}`;
      const d = new Date(iso + 'T00:00:00Z');
      if (isNaN(d.getTime())) return null;
      return d;
    }

    /**
     * Match 'N (days|weeks|months) ago' — EN/FR/RU. Returns a Date now-shifted
     * by the right number of days, or null. `now` is injectable for tests.
     */
    function matchRelativeAgo(text: string, now: Date): Date | null {
      const q = text.toLowerCase();
      // EN: "3 weeks ago", "10 days ago", "two months ago"
      // FR: "il y a 3 semaines", "il y a 10 jours"
      // RU: "3 недели назад", "10 дней назад"

      const enNum = q.match(/\b(\d+)\s+(day|week|month|year)s?\s+ago\b/);
      const enWord = q.match(/\b(one|two|three|four|five|six|seven|eight|nine|ten)\s+(day|week|month|year)s?\s+ago\b/);
      const frNum = q.match(/il y a\s+(\d+)\s+(jour|semaine|mois|an|année)s?/);
      const ruNum = q.match(/(\d+)\s+(дн|дней|недел|месяц|год|лет)\w*\s+назад/);

      const numberWords: Record<string, number> = {
        one: 1, two: 2, three: 3, four: 4, five: 5,
        six: 6, seven: 7, eight: 8, nine: 9, ten: 10,
      };

      let n: number | null = null;
      let unit: 'day' | 'week' | 'month' | 'year' | null = null;

      if (enNum) {
        n = parseInt(enNum[1], 10);
        unit = enNum[2] as typeof unit;
      } else if (enWord) {
        n = numberWords[enWord[1]];
        unit = enWord[2] as typeof unit;
      } else if (frNum) {
        n = parseInt(frNum[1], 10);
        const u = frNum[2];
        unit = u.startsWith('jour') ? 'day' : u.startsWith('semaine') ? 'week' : u === 'mois' ? 'month' : 'year';
      } else if (ruNum) {
        n = parseInt(ruNum[1], 10);
        const u = ruNum[2];
        unit = u.startsWith('дн') || u.startsWith('день') ? 'day' : u.startsWith('недел') ? 'week' : u.startsWith('месяц') ? 'month' : 'year';
      }

      if (n == null || unit == null) return null;

      const daysPerUnit: Record<typeof unit, number> = {
        day: 1, week: 7, month: 30, year: 365,
      } as const;
      const shifted = new Date(now.getTime() - n * daysPerUnit[unit] * 86_400_000);
      // Normalize to midnight UTC for consistency with absolute matches
      shifted.setUTCHours(0, 0, 0, 0);
      return shifted;
    }

    /**
     * Match EN/FR/RU month-name + day-of-month (e.g., "April 1", "1 avril", "1 апреля").
     * Year is inferred: if the resolved month+day has not yet happened in the
     * current calendar year, assume current year; otherwise use prior year.
     * Returns a Date or null.
     */
    function matchMonthDay(text: string, now: Date): Date | null {
      const months: Record<string, number> = {
        january: 0, february: 1, march: 2, april: 3, may: 4, june: 5,
        july: 6, august: 7, september: 8, october: 9, november: 10, december: 11,
        janvier: 0, février: 1, mars: 2, avril: 3, mai: 4, juin: 5,
        juillet: 6, août: 7, septembre: 8, octobre: 9, novembre: 10, décembre: 11,
        январ: 0, феврал: 1, март: 2, апрел: 3, мая: 4, июн: 5,
        июл: 6, август: 7, сентябр: 8, октябр: 9, ноябр: 10, декабр: 11,
      };
      const q = text.toLowerCase();
      // EN: "April 1", "April 1st", "April 1, 2026"
      const en = q.match(/\b(january|february|march|april|may|june|july|august|september|october|november|december)\s+(\d{1,2})(?:\s*,\s*(\d{4}))?/);
      // FR: "1 avril", "1er avril"
      const fr = q.match(/(\d{1,2})(?:er)?\s+(janvier|février|mars|avril|mai|juin|juillet|août|septembre|octobre|novembre|décembre)(?:\s+(\d{4}))?/);
      // RU: "1 апреля" (genitive, partial match via prefix)
      const ru = q.match(/(\d{1,2})\s+(январ|феврал|март|апрел|мая|июн|июл|август|сентябр|октябр|ноябр|декабр)/);

      let monthIdx: number | null = null;
      let day: number | null = null;
      let year: number | null = null;

      if (en) { monthIdx = months[en[1]]; day = parseInt(en[2], 10); year = en[3] ? parseInt(en[3], 10) : null; }
      else if (fr) { day = parseInt(fr[1], 10); monthIdx = months[fr[2]]; year = fr[3] ? parseInt(fr[3], 10) : null; }
      else if (ru) { day = parseInt(ru[1], 10); monthIdx = months[ru[2]]; }

      if (monthIdx == null || day == null) return null;

      if (year == null) {
        const thisYear = now.getUTCFullYear();
        const candidate = new Date(Date.UTC(thisYear, monthIdx, day));
        year = candidate.getTime() > now.getTime() ? thisYear - 1 : thisYear;
      }
      const d = new Date(Date.UTC(year, monthIdx, day));
      if (isNaN(d.getTime())) return null;
      return d;
    }

    function hasDateHeuristic(text: string): boolean {
      const q = text.toLowerCase();
      return DATE_HEURISTIC_KEYWORDS.some((kw) => q.includes(kw));
    }

    /**
     * Extract a query-about-date from the user's text. Regex/keyword fast-path
     * first (ISO date, N-units-ago, month-name+day). If the fast-path returns
     * null AND at least one DATE_HEURISTIC_KEYWORDS match is present, call
     * Haiku as a fallback classifier. If no heuristic match, return null
     * without any API call.
     *
     * Returns null on any failure (never throws).
     *
     * `now` is injectable for deterministic tests (default: `new Date()`).
     */
    export async function extractQueryDate(
      text: string,
      language?: string,
      now: Date = new Date(),
    ): Promise<Date | null> {
      // Fast-path 1: ISO date
      const iso = matchIsoDate(text);
      if (iso) return iso;

      // Fast-path 2: relative "N units ago"
      const rel = matchRelativeAgo(text, now);
      if (rel) return rel;

      // Fast-path 3: month-name + day-of-month
      const md = matchMonthDay(text, now);
      if (md) return md;

      // No regex match. Gate Haiku on heuristic keyword presence.
      if (!hasDateHeuristic(text)) return null;

      try {
        const response = await anthropic.messages.create({
          model: HAIKU_MODEL,
          max_tokens: 64,
          system: `You extract a single calendar date from a user query.
Output JSON only: { "date": "YYYY-MM-DD" } or { "date": null } if no specific date is identifiable.
The current date is ${now.toISOString().slice(0, 10)}.
"Last Tuesday" means the most recent past Tuesday. "Last week" means 7 days ago. "Yesterday" means 1 day ago.
Respond with ONLY the JSON object, no prose.`,
          messages: [{ role: 'user', content: text }],
        });

        const block = response.content.find((b: { type: string }) => b.type === 'text');
        if (!block || block.type !== 'text') return null;

        const parsed = JSON.parse((block as { type: 'text'; text: string }).text.trim());
        if (typeof parsed.date !== 'string') return null;
        const d = new Date(parsed.date + 'T00:00:00Z');
        if (isNaN(d.getTime())) return null;
        return d;
      } catch (error) {
        logger.warn(
          {
            text: text.slice(0, 80),
            language,
            error: error instanceof Error ? error.message : String(error),
          },
          'chris.date-extraction.haiku-error',
        );
        return null;
      }
    }
    ```
