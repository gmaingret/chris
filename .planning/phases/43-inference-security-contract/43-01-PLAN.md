<!-- The sanitizeSubstrateText contract, locked by this plan: -->

export function sanitizeSubstrateText(text: string): string;
// - Input: any string (no preconditions on length, content, or character set)
// - Output: a string where:
//     1. Every (^|\n)(#+\s) match is rewritten to insert a literal backslash
//        before the # sequence ("\## " not "##"). The hash itself remains
//        visible — Sonnet sees the content but can no longer parse it as a
//        section anchor (D-01).
//     2. Every triple-backtick fence (```) is rewritten to ''' (D-02).
// - Total: never throws. Empty string maps to empty string.
// - Idempotent: f(f(x)) === f(x). Escaped \\## does not re-escape (no
//   match for ^|\n followed by #+\s — backslash is not whitespace).

// Boundary rule (D-04 + D047): psychological-profile-prompt.ts MUST NOT
// import this from shared.ts (the operational vocabulary boundary forbids
// it). Re-implement the function body verbatim inside psychological-profile-
// prompt.ts. Duplication is intentional — 3 lines of regex is below the cost
// of a shared abstraction.
</interfaces>
