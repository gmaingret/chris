```typescript
// ── Pre-fire suppression (VOICE-04 — Plan 26-03) ───────────────────────────

/**
 * shouldSuppressVoiceNoteFire — Pitfall 9 mitigation.
 *
 * Returns true if today (local Europe/Paris day) already has
 * `RITUAL_SUPPRESS_DEPOSIT_THRESHOLD` (default 5) or more telegram-source
 * JOURNAL-mode Pensieve entries. The voice note ritual fire is then skipped
 * with the 'system_suppressed' outcome — Greg has clearly journaled
 * enough today that another prompt would feel redundant.
 *
 * Per CONTEXT.md D-26-05: query Pensieve directly (not conversations table),
 * matches Pitfall 9 mitigation language verbatim ("≥5 deposits today").
 *
 * Pensieve is the authoritative store (D035); the metadata.mode field is set
 * by JOURNAL-mode pensieve writers in src/chris/modes/journal.ts.
 */
export async function shouldSuppressVoiceNoteFire(now: Date): Promise<boolean> {
  const dayStart = dayBoundaryUtc(now, config.proactiveTimezone, 'start');
  const result = await db.select({
    count: sql<number>`COUNT(*)::int`,
  })
    .from(pensieveEntries)
    .where(and(
      eq(pensieveEntries.source, 'telegram'),
      gte(pensieveEntries.createdAt, dayStart),
      sql`${pensieveEntries.metadata}->>'mode' = 'JOURNAL'`,
    ));
  const count = result[0]?.count ?? 0;
  return count >= RITUAL_SUPPRESS_DEPOSIT_THRESHOLD;
}
```
