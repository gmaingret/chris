       ```typescript
       export async function updateAllOperationalProfiles(): Promise<void> {
         const startMs = Date.now();
         try {
           // 1. Load substrate ONCE (D-14)
           const substrate = await loadProfileSubstrate();

           logger.info(
             { entryCount: substrate.entryCount, episodicCount: substrate.episodicSummaries.length, decisionCount: substrate.decisions.length },
             'chris.profile.cron.start',
           );

           // 2. Promise.allSettled fan-out (D-21 — per-generator error isolation)
           const results = await Promise.allSettled([
             generateJurisdictionalProfile({ substrate }),
             generateCapitalProfile({ substrate }),
             generateHealthProfile({ substrate }),
             generateFamilyProfile({ substrate }),
           ]);

           // 3. Discriminated outcome aggregation (D-34)
           const counts = { updated: 0, skipped: 0, belowThreshold: 0, failed: 0 };
           for (const r of results) {
             if (r.status === 'fulfilled') {
               const o: ProfileGenerationOutcome = r.value;
               // Per-dimension log key already emitted INSIDE the generator (Plan 34-02);
               // here we count for the aggregate log
               switch (o.outcome) {
                 case 'profile_updated': counts.updated += 1; break;
                 case 'profile_skipped_no_change': counts.skipped += 1; break;
                 case 'profile_below_threshold': counts.belowThreshold += 1; break;
                 case 'profile_generation_failed': counts.failed += 1; break;
               }
             } else {
               // Generator threw before returning a discriminated outcome — emergency path (D-21 isolation)
               logger.warn(
                 { err: r.reason instanceof Error ? r.reason.message : String(r.reason) },
                 'chris.profile.profile_generation_failed',
               );
               counts.failed += 1;
             }
           }

           // 4. Aggregate cron-complete log (D-34)
           logger.info(
             { counts, durationMs: Date.now() - startMs },
             'chris.profile.cron.complete',
           );
         } catch (err) {
           // Outer try/catch belt-and-suspenders (CRON-01 — mirrors src/episodic/cron.ts:90-122)
           logger.error(
             { err: err instanceof Error ? err.message : String(err), durationMs: Date.now() - startMs },
             'profile.cron.error',  // LOWERCASE infra-error key per src/cron-registration.ts convention
           );
         }
       }
       ```

    The aggregate log key is `'chris.profile.cron.complete'` (D-34 verbatim). The infra-error key is `'profile.cron.error'` (lowercase, matching `episodic.cron.error` / `rituals.cron.error` / `sync.cron.error` convention at `src/cron-registration.ts:73,92,110,131,149` per RESEARCH.md line 973).
