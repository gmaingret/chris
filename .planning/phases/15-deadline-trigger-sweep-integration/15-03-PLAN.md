    ```typescript
    /**
     * Phase 15: sweep writes AWAITING_RESOLUTION before sending the accountability prompt.
     * Phase 16's engine PP#0 reads this row to route Greg's reply to the resolution handler.
     */
    export async function upsertAwaitingResolution(chatId: bigint, decisionId: string): Promise<void> {
      await db.insert(decisionCaptureState).values({
        chatId,
        stage: 'AWAITING_RESOLUTION',
        draft: {},
        decisionId,
      }).onConflictDoUpdate({
        target: decisionCaptureState.chatId,
        set: {
          stage: 'AWAITING_RESOLUTION',
          draft: {},
          decisionId,
          updatedAt: new Date(),
        },
      });
    }
    ```

    This uses the existing `decisionCaptureState` table. The `chatId` is the PK (enforces 1 active flow per chat). The `draft` field is an empty object since we're not capturing — we're awaiting resolution. The `decisionId` field links to the specific decision.

    New `runSweep()` structure:
    ```
    1. Global mute gate — isMuted() — if true, return { triggered: false, skippedReason: 'muted' }
    2. Initialize: let accountabilityResult: ChannelResult | undefined; let reflectiveResult: ChannelResult | undefined;

    3. === ACCOUNTABILITY CHANNEL (fires first per D-05) ===
       a. if (!await hasSentTodayAccountability(config.proactiveTimezone)):
          - const deadlineTrigger = createDeadlineTrigger();
          - const deadlineResult = await deadlineTrigger.detect();
          - if (deadlineResult.triggered):
            - const systemPrompt = ACCOUNTABILITY_SYSTEM_PROMPT.replace('{triggerContext}', deadlineResult.context);
            - Generate via anthropic.messages.create with SONNET_MODEL (per D-10)
            - If empty response, log error and skip
            - Else:
              - Extract decision ID from deadlineResult.evidence[0] (format: 'Decision ID: <uuid>')
              - const decisionId = deadlineResult.evidence![0]!.replace('Decision ID: ', '');
              - await upsertAwaitingResolution(BigInt(config.telegramAuthorizedUserId), decisionId);
              - await bot.api.sendMessage(config.telegramAuthorizedUserId, messageText);
              - await saveMessage(BigInt(config.telegramAuthorizedUserId), 'ASSISTANT', messageText, 'JOURNAL');
              - await setLastSentAccountability(new Date());
              - accountabilityResult = { triggered: true, triggerType: 'decision-deadline', message: messageText };
              - logger.info({ triggerType: 'decision-deadline', latencyMs: ... }, 'proactive.sweep.accountability.sent');

    4. === REFLECTIVE CHANNEL (independent per D-05) ===
       a. if (!await hasSentTodayReflective(config.proactiveTimezone)):
          - Run existing Phase 1 SQL triggers (silence + commitment) in Promise.all — SAME logic as current code
          - Phase 2 Opus if no SQL hits — SAME logic as current code
          - Winner selection by priority — SAME logic as current code
          - Generate via PROACTIVE_SYSTEM_PROMPT — SAME logic as current code
          - Send, save, setLastSentReflective(new Date()) — replaces old setLastSent
          - reflectiveResult = { triggered: true, triggerType: winner.triggerType, message: messageText };

    5. === Result assembly ===
       const triggered = !!(accountabilityResult?.triggered || reflectiveResult?.triggered);
       return {
         triggered,
         triggerType: accountabilityResult?.triggerType ?? reflectiveResult?.triggerType,
         message: accountabilityResult?.message ?? reflectiveResult?.message,
         skippedReason: triggered ? undefined : 'no_trigger',
         accountabilityResult,
         reflectiveResult,
       };
    ```

    **IMPORTANT: Error isolation between channels.** Wrap the accountability channel in try/catch — if the LLM call fails or upsertAwaitingResolution fails, log the error and continue to the reflective channel. The reflective channel must never be blocked by accountability failures.
