```ts
#!/usr/bin/env node
/**
 * scripts/backfill-episodic.ts — Phase 23 OPS-01
 *
 * Operator script for day-by-day episodic consolidation backfill.
 *
 * Usage:
 *   npx tsx scripts/backfill-episodic.ts --from 2026-04-13 --to 2026-04-17
 *
 * Behavior:
 *   - Sequentially calls runConsolidate(date) for each calendar date in [from, to] (inclusive, ascending).
 *   - 2-second delay between invocations (OPS-01 rate-limit spec).
 *   - Idempotent: relies on Phase 21 CONS-03 (pre-flight SELECT + ON CONFLICT DO NOTHING).
 *     Re-running is a no-op for already-summarized days.
 *   - Continue-on-error: a failed day logs the error and proceeds to the next day.
 *   - Exits 0 on full completion (including if every day was skipped or errored).
 *   - Exits 1 only on argument validation failure (not on per-day errors).
 */

import { parseArgs } from 'node:util';
import { runConsolidate } from '../src/episodic/consolidate.js';
import { logger } from '../src/utils/logger.js';

interface ParsedArgs {
  from: string;
  to: string;
}

function parseCliArgs(argv: string[]): ParsedArgs {
  const { values } = parseArgs({
    args: argv,
    options: {
      from: { type: 'string' },
      to: { type: 'string' },
    },
    strict: true,
    allowPositionals: false,
  });

  if (!values.from || !values.to) {
    throw new UsageError('Both --from and --to are required.');
  }

  const dateRegex = /^\d{4}-\d{2}-\d{2}$/;
  if (!dateRegex.test(values.from) || !dateRegex.test(values.to)) {
    throw new UsageError('--from and --to must be YYYY-MM-DD.');
  }

  const fromDate = new Date(values.from + 'T00:00:00Z');
  const toDate = new Date(values.to + 'T00:00:00Z');
  if (Number.isNaN(fromDate.getTime()) || Number.isNaN(toDate.getTime())) {
    throw new UsageError('--from and --to must parse as valid dates.');
  }
  if (fromDate.getTime() > toDate.getTime()) {
    throw new UsageError('--from must be <= --to.');
  }

  const now = Date.now();
  if (fromDate.getTime() > now || toDate.getTime() > now) {
    throw new UsageError('Backfill range must be in the past (--from and --to <= today).');
  }

  // Safety valve: reject ranges > 365 days to prevent accidental historical-Internet runs.
  const MAX_SPAN_MS = 365 * 24 * 60 * 60 * 1000;
  if (toDate.getTime() - fromDate.getTime() > MAX_SPAN_MS) {
    throw new UsageError('Backfill range > 365 days is not supported; split into multiple invocations.');
  }

  return { from: values.from, to: values.to };
}

class UsageError extends Error {}

function* iterateDates(from: string, to: string): Generator<string> {
  let current = new Date(from + 'T00:00:00Z');
  const end = new Date(to + 'T00:00:00Z');
  while (current.getTime() <= end.getTime()) {
    const y = current.getUTCFullYear();
    const m = String(current.getUTCMonth() + 1).padStart(2, '0');
    const d = String(current.getUTCDate()).padStart(2, '0');
    yield `${y}-${m}-${d}`;
    current = new Date(current.getTime() + 24 * 60 * 60 * 1000);
  }
}

const INTER_DAY_DELAY_MS = 2000; // OPS-01 spec: 2 seconds between days.

/**
 * Programmatic entrypoint (exported for the integration test).
 * Returns a summary { total, inserted, skipped, errored } for operator visibility.
 */
export async function runBackfill(
  from: string,
  to: string,
  opts?: { delayMs?: number },
): Promise<{ total: number; inserted: number; skipped: number; errored: number }> {
  const delayMs = opts?.delayMs ?? INTER_DAY_DELAY_MS;
  const totals = { total: 0, inserted: 0, skipped: 0, errored: 0 };
  let first = true;

  for (const dateStr of iterateDates(from, to)) {
    totals.total += 1;

    if (!first && delayMs > 0) {
      await new Promise((r) => setTimeout(r, delayMs));
    }
    first = false;

    try {
      // Construct a Date that represents midnight-of-the-day in UTC; Phase 21's runConsolidate
      // handles timezone-aware day-boundary resolution internally (per CONS-01).
      const dateObj = new Date(dateStr + 'T00:00:00Z');
      const result = await runConsolidate(dateObj);

      // Distinguish skipped vs inserted by Phase 21's return contract (CONS-03: `{ skipped: true }`).
      // If the return type uses a different marker, adapt; this is planner-verified against
      // src/episodic/consolidate.ts.
      if (result && typeof result === 'object' && 'skipped' in result && result.skipped === true) {
        totals.skipped += 1;
        logger.info({ date: dateStr, result: 'skipped' }, 'backfill.day');
      } else {
        totals.inserted += 1;
        logger.info(
          {
            date: dateStr,
            result: 'inserted',
            importance: (result as { importance?: number }).importance,
            topics: (result as { topics?: string[] }).topics,
          },
          'backfill.day',
        );
      }
    } catch (err) {
      totals.errored += 1;
      logger.error(
        { date: dateStr, result: 'error', error: err instanceof Error ? err.message : String(err) },
        'backfill.day',
      );
      // D-22: continue to next day. Do not rethrow.
    }
  }

  logger.info(totals, 'backfill.complete');
  return totals;
}

/**
 * CLI entrypoint — invoked when the script is run directly via `tsx scripts/backfill-episodic.ts ...`.
 */
async function main(): Promise<void> {
  try {
    const args = parseCliArgs(process.argv.slice(2));
    const result = await runBackfill(args.from, args.to);
    console.log(
      `backfill complete: ${result.inserted} inserted, ${result.skipped} skipped, ${result.errored} errored (total ${result.total} days)`,
    );
    process.exit(0);
  } catch (err) {
    if (err instanceof UsageError) {
      console.error(`backfill-episodic: ${err.message}`);
      console.error('Usage: backfill-episodic --from YYYY-MM-DD --to YYYY-MM-DD');
      process.exit(1);
    }
    console.error('backfill-episodic: unexpected error:', err);
    process.exit(1);
  }
}

// Only invoke main() when run as a script, not when imported for testing.
// import.meta.url check works in ESM; if project uses CJS, adapt the guard.
if (import.meta.url === `file://${process.argv[1]}`) {
  void main();
}
```

```ts
const FIXTURE_DATES = ['2026-04-13', '2026-04-14', '2026-04-15'] as const;

async function cleanupBackfillFixture() {

describe('OPS-01: backfill-episodic.ts integration', () => {
  beforeAll(async () => {

  beforeEach(() => {
```
