```typescript
import { db } from '../db/connection.js';
import { decisionEvents, decisions } from '../db/schema.js';
import { asc, eq } from 'drizzle-orm';

type DecisionRow = typeof decisions.$inferSelect;

// Columns stored as timestamptz in Postgres; when embedded in jsonb they round-trip
// as ISO 8601 strings but the live .select() returns Date objects. Rehydrate for
// deep-equal parity with the live projection.
const TIMESTAMPTZ_COLUMNS = [
  'createdAt',
  'updatedAt',
  'resolveBy',
  'resolvedAt',
  'reviewedAt',
  'withdrawnAt',
  'staleAt',
  'abandonedAt',
  'accuracyClassifiedAt',
] as const;

function rehydrateDates(snapshot: Record&lt;string, unknown&gt;): DecisionRow {
  const out: Record&lt;string, unknown&gt; = { ...snapshot };
  for (const col of TIMESTAMPTZ_COLUMNS) {
    const v = out[col];
    if (v === null || v === undefined) continue;
    if (v instanceof Date) continue;                 // already a Date (defensive)
    if (typeof v === 'string') out[col] = new Date(v);
    // numbers: unlikely (postgres.js returns ISO strings), but handle defensively
    else if (typeof v === 'number') out[col] = new Date(v);
  }
  // bigint chat_id round-trip: jsonb can't store bigint natively; postgres.js
  // serializes it as a string. Drizzle re-inflates via its bigint mode. If we
  // find a string where bigint is expected, coerce:
  if (typeof out['chatId'] === 'string') out['chatId'] = BigInt(out['chatId']);
  return out as DecisionRow;
}

/**
 * Replays `decision_events` for `id` in (created_at ASC, sequence_no ASC) order
 * and returns the snapshot of the last event, rehydrated to match the live
 * projection's types (Date for timestamptz; bigint for chat_id).
 *
 * Proves the append-only invariant (D-01, D-13): if this function deep-equals
 * the live `decisions` row, then every mutation of the projection row was
 * preceded by a faithful event append.
 */
export async function regenerateDecisionFromEvents(id: string): Promise&lt;DecisionRow | null&gt; {
  const events = await db
    .select()
    .from(decisionEvents)
    .where(eq(decisionEvents.decisionId, id))
    .orderBy(asc(decisionEvents.createdAt), asc(decisionEvents.sequenceNo));
  if (events.length === 0) return null;
  const last = events[events.length - 1]!;
  return rehydrateDates(last.snapshot as Record&lt;string, unknown&gt;);
}
```

Implementation notes:
- Ordering uses BOTH `createdAt` and `sequenceNo` (per special_constraints + regenerate.test.ts tiebreaker case).
- We return the last snapshot rather than merging — D-01 mandates full snapshots, so "replay" is literally "read the most recent event's snapshot."
- The rehydration list must include every timestamptz column from `decisions`; if schema adds new ones in future phases, this list updates too (consider a code comment pointing at TIMESTAMPTZ_COLUMNS).
- bigint chat_id handling: jsonb serializes bigint as string on the way in (depending on Drizzle version); we coerce on the way back. If the regenerate test shows `chatId` as `42n` in live but `"42"` in replay, this handles it. If the test doesn't seed `chatId`, this line is a no-op.
