export const VARIANCE_THRESHOLD = 0.4;            // WEEK-09 spec
export const INSUFFICIENT_DATA_THRESHOLD = 4;     // <4 snapshots → omit

export function computeStdDev(values: number[]): number;

export type WeeklyReviewContext = {
  summaries: EpisodicSummary[];

  wellbeingSnapshots: WellbeingSnapshot[];
  wellbeingVariance: { energy: number; mood: number; anxiety: number };

3. **`loadWeeklyReviewContext` implementation** — parallel fetch via `Promise.all`:
```typescript
const [summaries, resolvedDecisions, wellbeingSnapshots] = await Promise.all([
  getEpisodicSummariesRange(weekStart, weekEnd),
  db.select().from(decisions).where(and(
    eq(decisions.status, 'resolved'),
    gte(decisions.resolvedAt, weekStart),
    lte(decisions.resolvedAt, weekEnd),
  )),
  db.select().from(wellbeingSnapshots).where(and(
    gte(wellbeingSnapshots.snapshotDate, formatLocalDate(weekStart, config.proactiveTimezone)),
    lte(wellbeingSnapshots.snapshotDate, formatLocalDate(weekEnd, config.proactiveTimezone)),
  )),
]);
const wellbeingVariance = {
  energy: computeStdDev(wellbeingSnapshots.map((s) => s.energy)),
  mood:    computeStdDev(wellbeingSnapshots.map((s) => s.mood)),
  anxiety: computeStdDev(wellbeingSnapshots.map((s) => s.anxiety)),
};
const includeWellbeing = shouldIncludeWellbeing(wellbeingSnapshots);
return { summaries, resolvedDecisions, wellbeingSnapshots, wellbeingVariance, includeWellbeing };
```

4. **`shouldIncludeWellbeing` implementation**:
```typescript
export function shouldIncludeWellbeing(snapshots: WellbeingSnapshot[]): boolean {
  if (snapshots.length < INSUFFICIENT_DATA_THRESHOLD) {
    logger.info({ count: snapshots.length }, 'chris.weekly-review.wellbeing.insufficient-data');
    return false;
  }
  const v = {
    energy: computeStdDev(snapshots.map((s) => s.energy)),
    mood: computeStdDev(snapshots.map((s) => s.mood)),
    anxiety: computeStdDev(snapshots.map((s) => s.anxiety)),
  };
  return v.energy >= VARIANCE_THRESHOLD && v.mood >= VARIANCE_THRESHOLD && v.anxiety >= VARIANCE_THRESHOLD;
}
```
