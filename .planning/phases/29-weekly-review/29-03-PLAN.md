```sql
-- MD-29: Idempotency guards added so re-running this migration against a database
-- that already has the seed row does not fail. Drizzle's migrator tracks applied
-- migrations via __drizzle_migrations, so normal forward migration flows are
-- unaffected. The ON CONFLICT (name) DO NOTHING clause makes this safe for
-- cold-starts, manual re-applies (e.g. the raw-psql path in scripts/test.sh),
-- or recovery from a botched partial deploy.
--
-- Phase 29 (M009 v2.4) — Weekly Review seed row. Hand-authored per
-- CONTEXT.md D-09 (per-phase seed migration; Phase 26 owns 0007, Phase 27 owns 0008).
-- The drizzle-kit-generated meta/0009_snapshot.json + _journal.json entry
-- match this file's net schema effect (no schema change — INSERT-only).
--
-- next_run_at computation: next Sunday 20:00 Europe/Paris from migration apply time.
-- date_trunc('week', ...) returns Monday → +6 days = Sunday → +20 hours = 20:00.
-- CASE handles the "migration applied on Sunday after 20:00" edge case → schedule
-- for the FOLLOWING Sunday (NEXT week), so we never set a next_run_at in the past.

INSERT INTO "rituals" (name, type, next_run_at, config, enabled)
VALUES (
  'weekly_review',
  'weekly',
  CASE
    WHEN extract(dow FROM (now() AT TIME ZONE 'Europe/Paris')) = 0
         AND (now() AT TIME ZONE 'Europe/Paris')::time > '20:00'::time
    THEN ((date_trunc('week', now() AT TIME ZONE 'Europe/Paris') + interval '13 days 20 hours') AT TIME ZONE 'Europe/Paris')
    ELSE ((date_trunc('week', now() AT TIME ZONE 'Europe/Paris') + interval '6 days 20 hours') AT TIME ZONE 'Europe/Paris')
  END,
  '{"fire_at": "20:00", "fire_dow": 7, "skip_threshold": 2, "mute_until": null, "time_zone": "Europe/Paris", "prompt_set_version": "v1", "schema_version": 1}'::jsonb,
  true
)
ON CONFLICT (name) DO NOTHING;
```
