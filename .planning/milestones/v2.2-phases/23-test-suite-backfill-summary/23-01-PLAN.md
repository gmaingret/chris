```ts
describe('TEST-15 + TEST-16: 14-day fixture + Pearson importance correlation', () => {
  beforeAll(async () => {
    await cleanupFixture();
  });

  afterAll(async () => {
    // Leave the rows; downstream describe blocks may rely on them OR clean their own scope.
    // Safer: always clean at the end.
    await cleanupFixture();
  });

  it('TEST-16: Sonnet importance correlates with ground-truth at r > 0.7 across 14 fixture days', async () => {
    // Expand fixture dates aligned to FIXTURE_TZ.
    const fixtureDates = expandFixtureDates(); // 14 YYYY-MM-DD strings starting at FIXTURE_START_DATE

    // Assert length and tail coverage (TEST-15 enforcement inside TEST-16's body).
    expect(fixtureDates.length).toBe(14);
    expect(GROUND_TRUTH_LABELS.length).toBe(14);
    expect(GROUND_TRUTH_LABELS.some((l) => l <= 2)).toBe(true);
    expect(GROUND_TRUTH_LABELS.some((l) => l >= 9)).toBe(true);

    const assignedScores: number[] = [];

    for (let i = 0; i < 14; i++) {
      const date = fixtureDates[i]!;
      const label = GROUND_TRUTH_LABELS[i]!;
      const entries = buildEntriesForDay(i, label); // returns Array<{ content: string; epistemic_tag: string; ... }>

      // Seed Pensieve entries for this day (created_at aligned to the calendar date in FIXTURE_TZ).
      await seedPensieveEntries({ chatId: FIXTURE_CHAT_ID, date, tz: FIXTURE_TZ, entries });

      // Mock Sonnet to return an importance score close to but not equal to the label,
      // simulating a realistic Sonnet response that correlates with content but has noise.
      // For the fixture to PASS r > 0.7, each mock output's importance should be within ±1 of the label.
      const mockOutput: EpisodicSummarySonnetOutput = {
        summary: buildMockSummaryFor(i, label), // >= 50 chars narrative aligned to label severity
        importance: clampInRange(label + noiseForDay(i), 1, 10),
        topics: buildMockTopicsFor(i, label), // 2-4 topics
        emotional_arc: buildMockArcFor(i, label),
        key_quotes: entries.slice(0, 2).map((e) => e.content.slice(0, Math.min(e.content.length, 120))), // verbatim substrings
      };

      // messages.parse() returns { message: { content: [...] }, ... } — match the SDK's shape.
      // Phase 21's runConsolidate expects a parsed-output payload; match that.
      mockMessagesParse.mockResolvedValueOnce(mockParseResponseFor(mockOutput));

      // Advance mock clock to 23:00 of day `i` in FIXTURE_TZ (just before the cron would fire).
      const fireAt = dateAtLocalHour(date, FIXTURE_TZ, 23, 0);
      vi.setSystemTime(fireAt);

      // Invoke the engine.
      const result = await runConsolidate(new Date(date + 'T00:00:00')); // or however Phase 21 accepts dates

      // Assert a row was inserted; extract the assigned importance for correlation.
      expect(result).toBeDefined();
      const rows = await db.select().from(episodicSummaries).where(eq(episodicSummaries.summaryDate, date as unknown as Date));
      expect(rows).toHaveLength(1);
      assignedScores.push(rows[0]!.importance);
    }

    // Compute Pearson r.
    const r = pearsonCorrelation(assignedScores, [...GROUND_TRUTH_LABELS]);

    if (r <= 0.7) {
      // Build diagnostic breakdown per CONTEXT.md D-07 "fails loudly" requirement.
      const breakdown = assignedScores
        .map((a, idx) => `  day-${idx} (${fixtureDates[idx]}): assigned=${a}, label=${GROUND_TRUTH_LABELS[idx]}, delta=${Math.abs(a - GROUND_TRUTH_LABELS[idx]!)}`)
        .join('\n');
      throw new Error(
        `TEST-16 FAILED: Pearson r=${r.toFixed(3)} (expected > 0.7)\nPer-day breakdown:\n${breakdown}`,
      );
    }
    expect(r).toBeGreaterThan(0.7);
  });
});
```
