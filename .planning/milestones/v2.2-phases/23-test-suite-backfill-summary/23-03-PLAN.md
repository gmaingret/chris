/**
 * Phase 23 — /summary [YYYY-MM-DD] Telegram command (CMD-01).
 *
 * Usage patterns:
 *   /summary               → yesterday's episodic summary (in config.proactiveTimezone)
 *   /summary YYYY-MM-DD    → that date's episodic summary
 *   /summary <anything>    → usage help
 *
 * Edge cases:
 *   - Date in the future → "that date hasn't happened yet"
 *   - Date in the past with no row → "no summary for that date (you may not have written anything)"
 *   - DB error / null row where unexpected → localized generic error
 *
 * Security:
 *   - Inherits single-user Grammy auth middleware (D009).
 *   - chatId scoping is not required because episodic_summaries is a per-system table
 *     (Chris is single-user; D009). No per-chat filter on the query.
 */

import type { Context } from 'grammy';
import { getEpisodicSummary } from '../../pensieve/retrieve.js';
import { getLastUserLanguage } from '../../chris/language.js';
import { config } from '../../config.js';
import { logger } from '../../utils/logger.js';
import type { EpisodicSummary } from '../../episodic/types.js';

type Lang = 'English' | 'French' | 'Russian';

function langOf(raw: string | undefined): Lang {
  if (raw === 'French' || raw === 'Russian' || raw === 'English') return raw;
  return 'English';
}

const ISO_DATE = /^\d{4}-\d{2}-\d{2}$/;

// ── Localized message strings ───────────────────────────────────────────────

// These are authored to be short, direct, and consistent with Chris's voice
// (no flattery, no apology — D024/D027). Planner refines wording during
// implementation if other handlers have established idioms.

const MSG = {
  usage: {
    English: 'Use: /summary [YYYY-MM-DD]. No date = yesterday.',
    French: 'Utilisation : /summary [YYYY-MM-DD]. Sans date = hier.',
    Russian: 'Использование: /summary [YYYY-MM-DD]. Без даты — вчера.',
  },
  noRowPast: {
    English: (d: string) => `No summary for ${d}. You may not have written anything that day.`,
    French: (d: string) => `Pas de résumé pour le ${d}. Tu n'as peut-être rien écrit ce jour-là.`,
    Russian: (d: string) => `Нет сводки за ${d}. Возможно, в тот день ты ничего не записал.`,
  },
  noRowFuture: {
    English: (d: string) => `${d} hasn't happened yet.`,
    French: (d: string) => `Le ${d} n'est pas encore arrivé.`,
    Russian: (d: string) => `${d} ещё не наступило.`,
  },
  genericError: {
    English: 'I ran into trouble fetching that summary. Try again in a moment.',
    French: "J'ai eu un souci en récupérant ce résumé. Réessaie dans un instant.",
    Russian: 'Возникла проблема с получением этой сводки. Попробуй через мгновение.',
  },
  labels: {
    English: { summaryFor: 'Summary for', importance: 'importance', topics: 'Topics', arc: 'Emotional arc', quotes: 'Key moments' },
    French: { summaryFor: 'Résumé du', importance: 'importance', topics: 'Thèmes', arc: 'Arc émotionnel', quotes: 'Moments clés' },
    Russian: { summaryFor: 'Сводка за', importance: 'важность', topics: 'Темы', arc: 'Эмоциональная дуга', quotes: 'Ключевые моменты' },
  },
} as const;

function formatSummary(row: EpisodicSummary, lang: Lang): string {
  const L = MSG.labels[lang];
  const dateStr = typeof row.summary_date === 'string' ? row.summary_date : row.summary_date.toISOString().slice(0, 10);
  const lines = [
    `${L.summaryFor} ${dateStr} (${L.importance} ${row.importance}/10)`,
    '',
    row.summary,
    '',
    `${L.topics}: ${row.topics.join(', ')}`,
    `${L.arc}: ${row.emotional_arc}`,
  ];
  if (row.key_quotes.length > 0) {
    lines.push('', `${L.quotes}:`);
    for (const q of row.key_quotes) {
      lines.push(`- "${q}"`);
    }
  }
  return lines.join('\n');
}

// ── Date helpers in config.proactiveTimezone ────────────────────────────────

/** Returns YYYY-MM-DD for "yesterday" in the configured timezone. */
function yesterdayInTz(tz: string): string {
  // Compute "today" in the target tz, then subtract one day at calendar level.
  // Use Intl.DateTimeFormat with the explicit timeZone to build YYYY-MM-DD.
  const nowParts = new Intl.DateTimeFormat('en-CA', {
    timeZone: tz,
    year: 'numeric', month: '2-digit', day: '2-digit',
  }).formatToParts(new Date());
  const get = (t: string) => nowParts.find((p) => p.type === t)!.value;
  // Build a midnight-of-today-in-tz Date, subtract 24h, then reformat.
  const todayIso = `${get('year')}-${get('month')}-${get('day')}`;
  const todayMidnightUtc = Date.parse(`${todayIso}T00:00:00Z`);
  const yesterdayUtc = todayMidnightUtc - 24 * 60 * 60 * 1000;
  const y = new Date(yesterdayUtc);
  // Reformat in tz (handles DST crossings — the Intl path is the source of truth).
  const yParts = new Intl.DateTimeFormat('en-CA', {
    timeZone: tz,
    year: 'numeric', month: '2-digit', day: '2-digit',
  }).formatToParts(y);
  const gy = (t: string) => yParts.find((p) => p.type === t)!.value;
  return `${gy('year')}-${gy('month')}-${gy('day')}`;
}

/** Returns true if `YYYY-MM-DD` is strictly after today-in-tz. */
function isFutureDate(isoDate: string, tz: string): boolean {
  const nowParts = new Intl.DateTimeFormat('en-CA', {
    timeZone: tz,
    year: 'numeric', month: '2-digit', day: '2-digit',
  }).formatToParts(new Date());
  const get = (t: string) => nowParts.find((p) => p.type === t)!.value;
  const todayIso = `${get('year')}-${get('month')}-${get('day')}`;
  return isoDate > todayIso; // lexicographic compare works for YYYY-MM-DD strings
}

// ── Handler ──────────────────────────────────────────────────────────────────

export async function handleSummaryCommand(ctx: Context): Promise<void> {
  const chatId = ctx.chat?.id;
  if (chatId === undefined) return;

  const lang = langOf(getLastUserLanguage(chatId.toString()));
  const raw = ctx.message?.text ?? '';
  const after = raw.replace(/^\/summary(?:@\w+)?\s*/i, '').trim();

  let targetDate: string;

  if (after === '') {
    targetDate = yesterdayInTz(config.proactiveTimezone);
  } else if (ISO_DATE.test(after)) {
    targetDate = after;
  } else {
    await ctx.reply(MSG.usage[lang]);
    return;
  }

  // Future-date short-circuit (D-32).
  if (isFutureDate(targetDate, config.proactiveTimezone)) {
    await ctx.reply(MSG.noRowFuture[lang](targetDate));
    return;
  }

  try {
    const row = await getEpisodicSummary(new Date(targetDate + 'T00:00:00Z'));
    if (row === null || row === undefined) {
      await ctx.reply(MSG.noRowPast[lang](targetDate));
      return;
    }
    await ctx.reply(formatSummary(row, lang));
  } catch (err) {
    logger.warn(
      { chatId, targetDate, error: err instanceof Error ? err.message : String(err) },
      'summary.command.error',
    );
    await ctx.reply(MSG.genericError[lang]);
  }
}

/**
 * Phase 23 Plan 03 — /summary command handler integration test (CMD-01).
 *
 * Covers 5 input cases:
 *   a. /summary (no args) → yesterday's summary (row present)
 *   b. /summary 2026-04-15 → that date's summary (row present)
 *   c. /summary 2026-04-15 → past date with no row → "no summary for that date"
 *   d. /summary 2099-01-01 → future date → "hasn't happened yet"
 *   e. /summary garbage-input → usage help
 *
 * D-02: vi.setSystemTime ONLY.
 */

import { describe, it, expect, beforeAll, beforeEach, afterAll, afterEach, vi } from 'vitest';
import { eq, inArray } from 'drizzle-orm';
import { db } from '../../../db/connection.js';
import { episodicSummaries } from '../../../db/schema.js';
import { handleSummaryCommand } from '../summary.js';

const FIXTURE_CHAT_ID = 999_900_003;

function buildCtx(text: string): { captured: string[]; ctx: any } {
  const captured: string[] = [];
  const ctx = {
    chat: { id: FIXTURE_CHAT_ID },
    from: { id: FIXTURE_CHAT_ID },
    message: { text },
    reply: async (t: string) => {
      captured.push(t);
    },
  };
  return { captured, ctx };
}

async function seedSummary(date: string, importance: number): Promise<void> {
  await db.insert(episodicSummaries).values({
    summaryDate: date as unknown as Date, // Drizzle date column accepts ISO string
    summary: `Fixture summary for ${date}. Padded to at least fifty characters to satisfy the min-length constraint in place.`,
    importance,
    topics: ['test', 'fixture'],
    emotionalArc: 'flat',
    keyQuotes: [`verbatim quote from ${date}`],
    sourceEntryIds: [],
  });
}

async function cleanup() {
  await db.delete(episodicSummaries).where(inArray(episodicSummaries.summaryDate, [
    '2026-04-15',
    '2026-04-16',
    '2099-01-01',
    yesterdayInTzForTest(),
  ] as unknown as Date[]));
}

function yesterdayInTzForTest(): string {
  // Match the handler's exact logic; simple approximation sufficient here.
  const now = new Date();
  const y = new Date(now.getTime() - 24 * 60 * 60 * 1000);
  return y.toISOString().slice(0, 10);
}

describe('CMD-01: /summary handler', () => {
  beforeAll(async () => {
    await cleanup();
  });

  afterAll(async () => {
    await cleanup();
  });

  afterEach(async () => {
    await cleanup();
  });

  it('a) /summary (no args) returns yesterday summary when row exists', async () => {
    const yesterdayIso = yesterdayInTzForTest();
    await seedSummary(yesterdayIso, 5);

    const { captured, ctx } = buildCtx('/summary');
    await handleSummaryCommand(ctx);

    expect(captured).toHaveLength(1);
    expect(captured[0]).toContain(yesterdayIso);
    expect(captured[0]).toContain('5/10');
    expect(captured[0]).toContain('test, fixture');
  });

  it('b) /summary 2026-04-15 returns that date summary when row exists', async () => {
    await seedSummary('2026-04-15', 8);

    const { captured, ctx } = buildCtx('/summary 2026-04-15');
    await handleSummaryCommand(ctx);

    expect(captured).toHaveLength(1);
    expect(captured[0]).toContain('2026-04-15');
    expect(captured[0]).toContain('8/10');
  });

  it('c) /summary YYYY-MM-DD for past date with no row replies with a clear no-summary message', async () => {
    // Cleanup guarantees no row exists for 2026-04-16.
    const { captured, ctx } = buildCtx('/summary 2026-04-16');
    await handleSummaryCommand(ctx);

    expect(captured).toHaveLength(1);
    expect(captured[0].toLowerCase()).toMatch(/no summary|pas de résumé|нет сводки/);
    expect(captured[0]).toContain('2026-04-16');
    // Critically: the reply is NOT an error message (CMD-01 "not an error").

  });

  it('d) /summary for a future date replies with a "hasn\'t happened yet" message', async () => {

    await handleSummaryCommand(ctx);

    expect(captured).toHaveLength(1);
    expect(captured[0].toLowerCase()).toMatch(/hasn't happened|pas encore|ещё не наступило/);

  it('e) /summary with garbage input replies with usage help', async () => {

    expect(captured).toHaveLength(1);
    expect(captured[0].toLowerCase()).toMatch(/\/summary.*\[?yyyy-mm-dd\]?|utilisation|использование/);
  });
});
