  'decision_events',

    id: uuid('id').primaryKey().default(sql`gen_random_uuid()`),
    decisionId: uuid('decision_id').notNull().references(() =&gt; decisions.id),

export const decisionCaptureState = pgTable('decision_capture_state', {

**Notes on types:**
- `decisionEvents.sequenceNo` is declared as plain `bigint(...).notNull()` in schema.ts — NO `.generatedAlwaysAsIdentity()`. - `prediction text NOT NULL` because captures never land in `decisions` until the PREDICTION capture stage is complete; partial drafts live in `decision_capture_state.draft` (D-15), not in `decisions`. Phase 14 writes the row atomically only when all 5 slots are filled.
- Do NOT add CHECK constraints tied to non-open states here (per D-12 / special_constraints — defer those to the phase that writes them).
- Do NOT add a `.update()` helper for `decisionEvents` (special_constraints — append-only).
