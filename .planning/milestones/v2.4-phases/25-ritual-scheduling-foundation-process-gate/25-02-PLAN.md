```typescript
import { z } from 'zod';

/**
 * RitualConfig — bounded, validated config block stored in `rituals.config` jsonb.
 *
 * Per RIT-07: 8 named fields + schema_version. Strict mode rejects unknown fields
 * to prevent silent typos / future-fragility (Pitfall 5 prevention).
 *
 * Mirrors src/episodic/types.ts v3 schema pattern (NOT the v3/v4 dual pattern from
 * src/episodic/consolidate.ts:33-81 — that pattern is needed only at the
 * @anthropic-ai/sdk boundary, which Phase 25 does not touch).
 *
 * Forward-compat: all 4 cadences supported even though Phase 25 ships no monthly/
 * quarterly rituals (TS-10 — M013 may add them).
 */
export const RitualConfigSchema = z.object({
  fire_at: z.string().regex(/^([01]\d|2[0-3]):[0-5]\d$/, 'fire_at must be HH:mm'),
  fire_dow: z.number().int().min(1).max(7).optional(),
  prompt_bag: z.array(z.number().int().min(0).max(5)).max(6).optional(),
  skip_threshold: z.number().int().min(1).max(10),
  mute_until: z.string().datetime().nullable(),
  time_zone: z.string().min(1),
  prompt_set_version: z.string().min(1),
  schema_version: z.literal(1),
}).strict();

export type RitualConfig = z.infer<typeof RitualConfigSchema>;

export function parseRitualConfig(input: unknown): RitualConfig {
  return RitualConfigSchema.parse(input);
}

// ── Fire-outcome scaffolding (Phases 26-29 extend) ─────────────────────────

/**
 * RitualFireOutcome — discriminated outcomes the scheduler can emit.
 *
 * Phase 25 ships the union scaffold; per-outcome semantic enforcement and the
 * SKIP-01 RitualFireOutcome richer discriminated union ship in Phase 28.
 */
export type RitualFireOutcome =
  | 'fired'
  | 'caught_up'
  | 'muted'
  | 'race_lost'
  | 'in_dialogue'
  | 'config_invalid';

export interface RitualFireResult {
  ritualId: string;
  type: 'daily' | 'weekly' | 'monthly' | 'quarterly';
  fired: boolean;
  outcome: RitualFireOutcome;
  error?: unknown;
}
```

```typescript
import { describe, it, expect } from 'vitest';
import { computeNextRunAt } from '../cadence.js';
import type { RitualConfig } from '../types.js';

const baseConfig: RitualConfig = {
  fire_at: '21:00',
  skip_threshold: 3,
  mute_until: null,
  time_zone: 'Europe/Paris',
  prompt_set_version: 'v1',
  schema_version: 1,
};

describe('computeNextRunAt — daily cadence', () => {
  it('returns today 21:00 Paris when called before today\'s slot', () => {
    const now = new Date('2026-04-15T10:00:00.000Z'); // 12:00 Paris CEST
    const next = computeNextRunAt(now, 'daily', baseConfig);
    expect(next.toISOString()).toBe('2026-04-15T19:00:00.000Z'); // 21:00 Paris
  });

  it('advances to tomorrow when called after today\'s slot', () => {
    const now = new Date('2026-04-15T22:00:00.000Z'); // 00:00+1 Paris CEST
    const next = computeNextRunAt(now, 'daily', baseConfig);
    expect(next.toISOString()).toBe('2026-04-16T19:00:00.000Z');
  });
});

describe('computeNextRunAt — DST safety', () => {
  it('preserves wall-clock fire time across spring-forward (2026-03-29 Europe/Paris)', () => {

    const next = computeNextRunAt(now, 'daily', baseConfig);
    // 2026-03-29 21:00 Paris = 19:00 UTC (CEST, UTC+2)
    expect(next.toISOString()).toBe('2026-03-29T19:00:00.000Z');
  });

  it('preserves wall-clock fire time across fall-back (2026-10-25 Europe/Paris)', () => {
    const now = new Date('2026-10-24T19:00:00.000Z'); // 21:00 Paris CEST
```

```typescript
import type { RitualConfig } from './types.js';

// ── Cadence math (M009 Phase 25 RIT-08) ────────────────────────────────────

/**
 * Compute the next UTC fire instant for a ritual given current time + cadence + config.
 *
 * DST-safe: uses Luxon's wall-clock arithmetic (`plus({ days })` operates on
 * local time, not UTC ms) so a daily ritual configured for 21:00 Paris fires
 * at 21:00 Paris on both 2026-03-29 (spring-forward, 23h day) and 2026-10-25
 * (fall-back, 25h day).
 *
 * Mirrors the canonical `dayBoundaryUtc` pattern in src/episodic/sources.ts:74-83.
 *
 * Anchored to wall-clock target time (config.fire_at), NEVER to last_run_at —
 * prevents Pitfall 3 cadence drift (config.fire_at='21:00' must stay 21:00 even
 * if the cron sweep that fires the ritual runs at 10:00).
 *
 * Signature locked per CONTEXT.md D-09 (2026-04-26): cadence is passed
 * explicitly as the second parameter (sourced from rituals.type enum column)
 * rather than living inside RitualConfig (jsonb column) — keeps the schema
 * bounded, prevents column/schema duplication, and makes the dependency
 * explicit at every call site (which already has ritual.type in scope).
 */
export function computeNextRunAt(
  now: Date,
  cadence: 'daily' | 'weekly' | 'monthly' | 'quarterly',
  config: RitualConfig,
): Date {
  const tz = config.time_zone;
  const [hh, mm] = config.fire_at.split(':').map((s) => Number(s));

  // Today's local fire slot (in tz)
  let target = DateTime.fromJSDate(now, { zone: tz })
    .startOf('day')
    .set({ hour: hh!, minute: mm!, second: 0, millisecond: 0 });

  // If today's slot is already past, start from tomorrow
  if (target.toJSDate() <= now) {
    target = target.plus({ days: 1 });
  }

  // Cadence advancement
  switch (cadence) {
    case 'daily':
      // target is already today (future) or tomorrow — return as-is
      break;
    case 'weekly': {
      // Advance to the next ISO weekday matching fire_dow (1=Mon..7=Sun)
      const targetDow = config.fire_dow ?? 7; // Sunday default
      const daysToAdd = ((targetDow - target.weekday + 7) % 7) || 7;
      target = target.plus({ days: daysToAdd });
      break;
    }
    case 'monthly':
      target = target.plus({ months: 1 });
      break;
    case 'quarterly':
      target = target.plus({ months: 3 });
      break;
  }

  return target.toUTC().toJSDate();
}
```

```typescript
import { and, eq, isNull, lt, or, sql } from 'drizzle-orm';
import { db } from '../db/connection.js';
import { rituals } from '../db/schema.js';

// ── Atomic ritual-fire concurrency primitive (M009 Phase 25 RIT-10) ────────

/**
 * tryFireRitualAtomic — atomic check-and-set for ritual fire.
 *
 * Mirrors M007 D-28 optimistic-concurrency pattern from
 * src/decisions/lifecycle.ts (transitionDecision). Postgres row-level locking
 * + WHERE-clause predicate gives atomic check-and-set in one round-trip — no
 * SELECT-then-UPDATE race window.
 *
 * Race semantics: two concurrent invocations targeting the same ritual ID
 * under the same lastObserved value will produce exactly ONE returned row
 * (the winner) and ONE empty result (the loser whose WHERE-guard predicate
 * fails because the winner's UPDATE already advanced last_run_at past
 * lastObserved).
 *
 * Phase 25 ships the atomicity primitive; Phase 28 layers ritual_fire_events
 * append-only audit on top. Phase 25 itself does NOT need db.transaction()
 * because there is no event-INSERT to atomically pair with.
 */
export interface TryFireResult {
  fired: boolean;
  row?: typeof rituals.$inferSelect;
}

export async function tryFireRitualAtomic(
  ritualId: string,
  lastObserved: Date | null,
  newNextRunAt: Date,
): Promise<TryFireResult> {
  const updated = await db
    .update(rituals)
    .set({ lastRunAt: new Date(), nextRunAt: newNextRunAt })
    .where(
      and(
        eq(rituals.id, ritualId),
        or(
          isNull(rituals.lastRunAt),
          lastObserved ? lt(rituals.lastRunAt, lastObserved) : sql`true`,
        ),
      ),
    )
    .returning();

  if (updated.length === 0) {
    // Race lost — peer sweep already fired this ritual, OR our lastObserved is stale.
    return { fired: false };
  }
  return { fired: true, row: updated[0] };
}
```
